class Node
{
    public:
    int data;  // data 
    Node* next; //pointer to that next node
}; //Single Linklist class

Node* push(Node** head,int val)
{
    Node* temp= new Node();
    temp->next= (*head);
    temp->data=val;
    (*head)=temp;
    
}
void display(Node* head)
{
    Node* temp=head;

}
